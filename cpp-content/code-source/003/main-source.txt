void my_SendF11Command ()
{

	INPUT inputs[2] = {};

	for ( int i = 0 ; i < 2 ; i++ )
	{
		inputs[i].type = INPUT_KEYBOARD;
		inputs[i].ki.wVk = VK_F11;
		inputs[i].ki.time = 0;
		inputs[i].ki.dwExtraInfo = 0;
	}

	inputs[0].ki.dwFlags = 0;
	inputs[1].ki.dwFlags = KEYEVENTF_KEYUP;

	SendInput( 2 , inputs , sizeof(INPUT) );

}
