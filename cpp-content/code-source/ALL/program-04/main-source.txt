// SEND SHORTCUT
void my_SendCommand_Function(UINT key)
{

	INPUT inputs[2] = {};

	inputs[0].ki.wVk = key; inputs[0].ki.dwFlags = 0;
	inputs[1].ki.wVk = key; inputs[1].ki.dwFlags = KEYEVENTF_KEYUP;


	for (int counter = 0; counter < 2; counter++)
	{
		inputs[counter].type = INPUT_KEYBOARD;
		inputs[counter].ki.wScan = 0;
		inputs[counter].ki.time = 0;
		inputs[counter].ki.dwExtraInfo = 0;
	}

	SendInput(2, inputs, sizeof(INPUT));

}
